\sqlrow[fldsep= &,rowsep= \\]{
SELECT c.name
,      i.nr
,      i.invoice_date
,      IFNULL(i.pay_date, "-")
,      i.reminders
,      IFNULL(i.last_reminder, "-")
FROM   invoice i
LEFT JOIN customer c
     ON c.nr = i.cust_nr
ORDER BY i.invoice_date
}\\\hline
\end{tabular}
